    \begin{algorithm}[H]
	\caption{Algorithm:Naive Defensive algorithm}
    	\label{alg:q-learning}
	\begin{algorithmic}
	\Function{Get Action}{$\vars{S as state}$}
		\State $\vars{Functional Cards}$ $\gets$ $\vars{S.Playable}$ $\cap$ $\vars{S.Functional}$
		\State $\vars{Normal Cards}$ $\gets$ $\vars{S.Playable}$ $\cap$ ($\vars{S}$ $\setminus$ $\vars{S.Functional}$)
		\If{$\vars{Normal Cards}$ $\textbf{is not}$ $\varnothing$}
			\State \Return Random.GetOne($\vars{Normal Cards}$)
		\ElsIf{$\vars{Functional Cards}$ $\textbf{is not}$ $\varnothing$}
			\State \Return Random.GetOne($\vars{Functional Cards}$)
		\Else
			\State \Return $\vars{Take card from deck}$
		\EndIf
	\EndFunction
	\end{algorithmic}
